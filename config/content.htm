HW Triangle Star


向ChatGPT詢問20種五階金字塔的寫法




幫我用python寫一個五階金字塔，由*組成，給我20種不同寫法


1. 基本迴圈


for i in range(1, 6):
    print(' ' * (5 - i) + '*' * (2 * i - 1))




2. 使用字符串連接


for i in range(1, 6):
    spaces = ' ' * (5 - i)
    stars = '*' * (2 * i - 1)
    print(spaces + stars)



3. 使用格式化字串


for i in range(1, 6):
    print(f"{' ' * (5 - i)}{'*' * (2 * i - 1)}")



4. 列表生成式


pyramid = [' ' * (5 - i) + '*' * (2 * i - 1) for i in range(1, 6)]
print('\n'.join(pyramid))



5. 使用函數


def print_pyramid(n):
    for i in range(1, n + 1):
        print(' ' * (n - i) + '*' * (2 * i - 1))

print_pyramid(5)



6. 反向迴圈


for i in range(5, 0, -1):
    print(' ' * (5 - i) + '*' * (2 * i - 1))



7. 使用 join


for i in range(1, 6):
    print(''.join([' ' * (5 - i), '*' * (2 * i - 1)]))



8. 使用 map


for i in range(1, 6):
    print(''.join(map(str, [' ' * (5 - i), '*' * (2 * i - 1)])))



9. 使用遞迴


def print_recursive(n, current=1):
    if current > n:
        return
    print(' ' * (n - current) + '*' * (2 * current - 1))
    print_recursive(n, current + 1)

print_recursive(5)



10. 使用 while 迴圈


i = 1
while i <= 5:
    print(' ' * (5 - i) + '*' * (2 * i - 1))
    i += 1



11. 使用 str.format


for i in range(1, 6):
    print('{}{}'.format(' ' * (5 - i), '*' * (2 * i - 1)))



12. 使用 f-string 和 list comprehension


pyramid = [f"{' ' * (5 - i)}{'*' * (2 * i - 1)}" for i in range(1, 6)]
print('\n'.join(pyramid))



13. 使用 print 的多參數


for i in range(1, 6):
    print(' ' * (5 - i), '*' * (2 * i - 1), sep='')



14. 使用 zip


spaces = [' ' * (5 - i) for i in range(1, 6)]
stars = ['*' * (2 * i - 1) for i in range(1, 6)]
for s, st in zip(spaces, stars):
    print(s + st)



15. 使用 reduce


from functools import reduce
pyramid = reduce(lambda acc, i: acc + [' ' * (5 - i) + '*' * (2 * i - 1)], range(1, 6), [])
print('\n'.join(pyramid))



16. 使用 numpy


import numpy as np
pyramid = np.array([' ' * (5 - i) + '*' * (2 * i - 1) for i in range(1, 6)])
print('\n'.join(pyramid))



17. 使用類別


class Pyramid:
    def __init__(self, height):
        self.height = height

    def draw(self):
        for i in range(1, self.height + 1):
            print(' ' * (self.height - i) + '*' * (2 * i - 1))

p = Pyramid(5)
p.draw()



18. 使用生成器


def generate_pyramid(n):
    for i in range(1, n + 1):
        yield ' ' * (n - i) + '*' * (2 * i - 1)

for line in generate_pyramid(5):
    print(line)



19. 使用 itertools


from itertools import repeat
for i in range(1, 6):
    print(''.join(repeat(' ', 5 - i) + repeat('*', 2 * i - 1)))



20. 使用 string 模組


import string
for i in range(1, 6):
    print(string.whitespace * (5 - i) + '*' * (2 * i - 1))
